If any child requirement is met, then the requirement is met.

<pre>
REQUIREMENT
{
    name = Any
    type = Any

    REQUIREMENT
    {
        name = ReachSpace
        type = ReachSpace
    }

    REQUIREMENT
    {
        name = TechResearched
        type = TechResearched

        tech = basicRocketry
    }
}
</pre>
