If all child requirements are met, then the requirement is met.

<pre>
REQUIREMENT
{
    name = All
    type = All

    REQUIREMENT
    {
        name = ReachSpace
        type = ReachSpace
    }

    REQUIREMENT
    {
        name = TechResearched
        type = TechResearched

        tech = basicRocketry
    }
}
</pre>
